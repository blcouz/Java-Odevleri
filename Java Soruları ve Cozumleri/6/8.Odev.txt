    static Boolean BuyukUnluUyumunaUyarmi(String kelime) {//1.soru
        char[] kalinSesliler = {'a', 'ı', 'o', 'u', 'A', 'I', 'O', 'U'};
        char[] inceSesliler = {'e', 'i', 'ö', 'ü', 'E', 'İ', 'Ö', 'Ü'};
        boolean kalinVarmi = false;
        boolean inceVarmi = false;
        for (int i = 0; i < kelime.length(); i++) {
            if (!kalinVarmi) {
                for (int j = 0; j < kalinSesliler.length; j++) {
                    if (kalinSesliler[j] == kelime.charAt(i)) kalinVarmi = true;
                }
            }
            if (!inceVarmi) {
                for (int j = 0; j < inceSesliler.length; j++) {
                    if (inceSesliler[j] == kelime.charAt(i)) inceVarmi = true;
                }
            }
            if (kalinVarmi == true & inceVarmi == true) break;
        }
        if (kalinVarmi == true & inceVarmi == true) return false;
        else return true;
    }

    static void OHIleBitiyormu(String[] list) {//3.soru
        for (int i = 0; i < list.length; i++) {
            if (list[i].substring(list[i].length() - 2, list[i].length()).equals("OH"))
                System.out.println(list[i] + ", OH ile bitiyor");
            else System.out.println(list[i] + ", OH ile bitmiyor.");
        }
    }

    static int KacSesliHarfVar(String cumle) {//4.soru
        List<Character> sesliHarfler;
        Character[] gcc = {'a', 'ı', 'o', 'u', 'A', 'I', 'O', 'U', 'e', 'i', 'ö', 'ü', 'E', 'İ', 'Ö', 'Ü'};
        sesliHarfler = Arrays.asList(gcc);
        int sesliHarfSayisi = 0;
        for (int i = 0; i < cumle.length(); i++) {
            if (sesliHarfler.contains(cumle.charAt(i))) sesliHarfSayisi++;
        }
        return sesliHarfSayisi;
    }

    static int KacVeVar(String cumle) {//5.soru
        int veAdedi = 0;
        for (int i = 0; i < cumle.length() - 1; i++) {
            if (cumle.substring(i, i + 2).toLowerCase().equals("ve")) veAdedi++;
        }
        return veAdedi;
    }

    static String AltinciSoru(String cumle) {//6.soru
        String sonuc = "";
        for (int i = 0; i < cumle.length(); i += 3) {
            if (i < cumle.length()) {
                sonuc += cumle.charAt(i);
            }
        }
        return sonuc;
    }

class EnUzunKelimeBul {//2.soru
    static class KelimeVeUzunluk {
        String kelime;
        String uzunluk;
    }

    KelimeVeUzunluk EnUzunKelimeBul() {
        Scanner scn = new Scanner(System.in);
        System.out.print("Cümle giriniz : ");
        String cumle = scn.nextLine();
        KelimeVeUzunluk snc = new KelimeVeUzunluk();
        String[] kelimeler = cumle.trim().split(" ");
        String enUzunKelime = "";
        int enUzunKelimeUzunlugu = 0;
        for (int i = 0; i < kelimeler.length; i++) {
            if (kelimeler[i].trim().length() > enUzunKelimeUzunlugu) {
                enUzunKelime = kelimeler[i].trim();
                enUzunKelimeUzunlugu = enUzunKelime.length();
            }
        }
        snc.kelime = "En Uzun Kelime : " + enUzunKelime + "\n";
        snc.uzunluk = "Uzunluk : " + enUzunKelimeUzunlugu;
        return snc;
    }
