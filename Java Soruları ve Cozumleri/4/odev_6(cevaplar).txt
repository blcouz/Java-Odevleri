public static void main(String[] args) {
        Scanner scn = new Scanner(System.in);
        System.out.print("İşlem giriniz. ");
        String metin = scn.nextLine();
        ParantezHatasiVarmi(metin);

    }
    public static Boolean ParantezHatasiVarmi(String metin){
        Stack<Character> yigin = new Stack<>();
        for (int i = 0; i < metin.length(); i++) {
            Character harf = metin.charAt(i);
            if( harf =='{'|harf =='}'|harf =='('|harf ==')'|harf =='['|harf ==']') {
                if (yigin.isEmpty()) {
                    yigin.push(harf);
                } else {
                    if (yigin.peek() == '{') {
                        if (harf == '}') yigin.pop();
                        else yigin.push(harf);
                    } else if (yigin.peek() == '(') {
                        if (harf == ')') yigin.pop();
                        else yigin.push(harf);
                    } else if (yigin.peek() == '[') {
                        if (harf == ']') yigin.pop();
                        else yigin.push(harf);
                    }
                }
            }
        }
        return yigin.isEmpty();
    }
